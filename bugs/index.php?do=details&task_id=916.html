Hi,

I have recently started using "slim" as login manager with the following

~/.xinitrc:

#!/bin/sh

PATH=$HOME/bin:/usr/local/texlive/2009/bin/i386-unknown-freebsd8.0:/sbin:/bin:/usr/sbin:/usr/bin:/usr/games:/usr/local/sbin:/usr/local/bin:/usr/X11R6/bin:/usr/local/kde4/bin/;
export PATH

cd $HOME

xrdb -load $HOME/.Xresources
xmodmap $HOME/.xmodmaprc

xscreensaver -nosplash &

if [ -x "$(which gpg-agent)" ]; then
if test -f $HOME/.gpg-agent-info && kill -0 $(cut -d: -f 2 $HOME/.gpg-agent-info) 2>/dev/null; then
GPG_AGENT_INFO=$(cat $HOME/.gpg-agent-info)
export GPG_AGENT_INFO
else
eval $(gpg-agent --daemon)
echo $GPG_AGENT_INFO >$HOME/.gpg-agent-info
fi
fi

GPG_TTY=$(tty)
export GPG_TTY

case "$1" in
awesome) exec awesome ;;
fluxbox) exec startfluxbox ;;
*) exec awesome ;;
esac

~/.Xresources:

URxvt*geometry: 120x60
URxvt*font: -xos4-terminus-medium-r-normal-*-16-*-*-*-*-*-iso10646-1
URxvt*saveLines: 10000
URxvt*background: Black
URxvt*foreground: White
URxvt*scrollKey: true
URxvt*scrollTtyOutput: false
URxvt*scrollTtyKeypress: true

in my rc.lua I have:

terminal = "urxvt"

and:

awful.key({ modkey, }, "Return", function () awful.util.spawn(terminal) end),

but when I start urxvt with MOD + Return, it does not honour the xrdb setting for font. If I start urxvt from there (from the one that I have started with MOD+Return), it honours the xrdb setting. Here is the output of xrdb -query:

URxvt*geometry: 120x60
URxvt*font: -xos4-terminus-medium-r-normal-*-16-*-*-*-*-*-iso10646-1
URxvt*saveLines: 10000
URxvt*background: Black
URxvt*foreground: White
URxvt*scrollKey: true
URxvt*scrollTtyOutput: false
URxvt*scrollTtyKeypress: true

so everything looks ok, except that the instances started via awesome do not honour these settings.

Can you please help me?
