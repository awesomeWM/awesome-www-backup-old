client.connect_signal(
"manage", function (c, startup)

....

if c.instance:sub(0,11) == "sun-awt-X11"
and awful.client.floating.get(c) then
c.size_hints_honor = false
awful.client.moveresize(0,0,1,1,c)
awful.client.moveresize(0,0,-1,-1,c)
c.size_hints_honor = true
else
c.size_hints_honor = false
end
