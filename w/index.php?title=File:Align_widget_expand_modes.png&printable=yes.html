This is a screen shot depicting the basics of what happens when different expand modes are set.

Code to generate: 


-- {{{ Align test
align_wibox_test = awful.wibox({ position = "bottom", screen = 1, height=99 })

local textboxes = {}
local backgrounds = {}

local layout = wibox.layout.flex.vertical()

    local align_none = wibox.layout.align.horizontal()
        align_none:set_expand("none")
        backgrounds.none_left = wibox.widget.background()
            backgrounds.none_left:set_bg("AA3300")
            textboxes.none_left = wibox.widget.textbox()
                textboxes.none_left:set_text("none_left")
                textboxes.none_left:set_align("center")
            backgrounds.none_left:set_widget(textboxes.none_left)
        align_none:set_left(backgrounds.none_left)
        backgrounds.none_right = wibox.widget.background()
            backgrounds.none_right:set_bg("006622")
            textboxes.none_right = wibox.widget.textbox()
                textboxes.none_right:set_text("none_right Larger demonstration")
                textboxes.none_right:set_align("center")
            backgrounds.none_right:set_widget(textboxes.none_right)
        align_none:set_right(backgrounds.none_right)
        backgrounds.none_center = wibox.widget.background()
            backgrounds.none_center:set_bg("1133BB")
            textboxes.none_center = wibox.widget.textbox()
                textboxes.none_center:set_text("none_center This is the somewhat larger center text box.")
                textboxes.none_center:set_align("center")
            backgrounds.none_center:set_widget(textboxes.none_center)
        align_none:set_middle(backgrounds.none_center)
    layout:add(align_none)

    local align_inside = wibox.layout.align.horizontal()
        align_inside:set_expand("inside")
        backgrounds.inside_left = wibox.widget.background()
            backgrounds.inside_left:set_bg("277600")
            textboxes.inside_left = wibox.widget.textbox()
                textboxes.inside_left:set_text("inside_left")
                textboxes.inside_left:set_align("center")
            backgrounds.inside_left:set_widget(textboxes.inside_left)
        align_inside:set_left(backgrounds.inside_left)
        backgrounds.inside_right = wibox.widget.background()
            backgrounds.inside_right:set_bg("0028FF")
            textboxes.inside_right = wibox.widget.textbox()
                textboxes.inside_right:set_text("inside_right Larger demonstration")
                textboxes.inside_right:set_align("center")
            backgrounds.inside_right:set_widget(textboxes.inside_right)
        align_inside:set_right(backgrounds.inside_right)
        backgrounds.inside_center = wibox.widget.background()
            backgrounds.inside_center:set_bg("AA3331")
            textboxes.inside_center = wibox.widget.textbox()
                textboxes.inside_center:set_text("inside_center This is the somewhat larger center text box.")
                textboxes.inside_center:set_align("center")
            backgrounds.inside_center:set_widget(textboxes.inside_center)
        align_inside:set_middle(backgrounds.inside_center)
    layout:add(align_inside)

    local align_outside = wibox.layout.align.horizontal()
        align_outside:set_expand("outside")
        backgrounds.outside_left = wibox.widget.background()
            backgrounds.outside_left:set_bg("0000A4")
            textboxes.outside_left = wibox.widget.textbox()
                textboxes.outside_left:set_text("outside_left")
                textboxes.outside_left:set_align("center")
            backgrounds.outside_left:set_widget(textboxes.outside_left)
        align_outside:set_left(backgrounds.outside_left)
        backgrounds.outside_right = wibox.widget.background()
            backgrounds.outside_right:set_bg("BB0016")
            textboxes.outside_right = wibox.widget.textbox()
                textboxes.outside_right:set_text("outside_right Larger demonstration")
                textboxes.outside_right:set_align("center")
            backgrounds.outside_right:set_widget(textboxes.outside_right)
        align_outside:set_right(backgrounds.outside_right)
        backgrounds.outside_center = wibox.widget.background()
            backgrounds.outside_center:set_bg("128521")
            textboxes.outside_center = wibox.widget.textbox()
                textboxes.outside_center:set_text("outside_center This is the somewhat larger center text box.")
                textboxes.outside_center:set_align("center")
            backgrounds.outside_center:set_widget(textboxes.outside_center)
        align_outside:set_middle(backgrounds.outside_center)
    layout:add(align_outside)

align_wibox_test:set_widget(layout)

-- }}}
